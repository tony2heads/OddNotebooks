import pylab

from scipy import *

"""

This is the frequency version of blackbody enery density

"""



h=6.6244e-34 # Planck constant in Js

c=2.998e8     # speed of light in m/s

k=1.38e-23    #Boltzmann constant in J/K%



def blackbody(nu,T): # nu is frequency (Hz) , T is temperature (K)

    u_c=(8*pi*nu**2*k*T)/c**3 #classical part of energy density

    x=(h*nu)/(k*T)

    eb=x/(exp(x)-1 ) #Einstein-Bose correction factor - no units

    u_q=u_c*eb  # energy density

    S=u_q*c/4.0 # radiation

    return S
